class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_map<char,int>m;
        int maxlen=0,l=0;
        for(int r=0;r<s.size();r++)
        {
            char c=s[r];
            if(m.find(c)!=m.end()&&m[c]>=l)
            {
                l=m[c]+1;
            }
            m[c]=r;
            maxlen=max(maxlen,r-l+1);
        }
        return maxlen;
        }
};
